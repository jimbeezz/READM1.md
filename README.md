    primes = [True] * (limit + 1) 
    primes[0] = primes[1] = False # 0 –∏ 1 –Ω–µ –ø—Ä–æ—Å—Ç—ã–µ

    for num in range(2, int(limit**0.5) + 1):
        if primes[num]: # –ï—Å–ª–∏ —á–∏—Å–ª–æ –ø—Ä–æ—Å—Ç–æ–µ
            for multiple in range(num * num, limit + 1, num):

    return prime_numbers
